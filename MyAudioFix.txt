# This is my way to force my rates and latency but keep in mind i dont look for low latency switches.
# My focus is to have less CPU usage for audio and have a smooth playback instead of dynamic latency.
# Pipwire-pulse tends to switch to 10ms or less when he wants.
# A reminder that default.clock.quantum=2048 is the max unless you set clock.power-of-two-quantum=false as it calculates it with a power of two (1024-2048-4096-8192) the minimum i believe is 0 auto but you might face a bunch of audio crackling bellow 1024...


/home/baz/.config/pipewire/pipewire.conf.d/pwrate.conf


{
  "context.properties": {
    "default.clock.rate": "44100",
    "default.clock.allowed-rates": [
      "44100"
    ],
    "default.clock.quantum": "3072",
    "default.clock.quantum-limit": "8192",
    "default.clock.min-quantum": "3072",
    "default.clock.max-quantum": "8192",
    "link.max-buffers": "64",
    "log.level": "0",
    "clock.power-of-two-quantum": false,
    "default.clock.quantum-floor": "64"
  }
}

##### For Firefox to work and some other apps as they will have zero sound without it....

/home/baz/.config/pipewire/pipewire-pulse.conf.d/Firefox-rules.conf


pulse.rules = [
    {
        # skype does not want to use devices that don't have an S16 sample format.
        matches = [
             #{ application.process.binary = "teams" }
             #{ application.process.binary = "teams-insiders" }
             { application.process.binary = "firefox" }
        ]
        actions = { quirks = [ force-s16-info ] }
    }
    {
        # speech dispatcher asks for too small latency and then underruns.
        matches = [ { application.name = "~firefox*" } ]
        actions = {
            update-props = {
                pulse.min.req          = 0/44100     # 21ms
                pulse.min.quantum      = 0/44100     # 21ms
                default.clock.rate = 44100
            }
        }
    }
]






###### For wireplumber rules as this is the magic place to make that 16Bit forced to work.
# You can add multiple formats in [] if you want but i prefer 44.1khz and audio format also can be changed to whatever you want as S16LE S24LE S32LE or even S8LE for 8Bit in case its that old.
Also keep in mind that latency grows a bit when rates are changed so keep a tweaking on api.alsa.period-size= the higher the audio.rate the latency also grows to compensate.
I found mine working best with this.



/home/baz/.config/wireplumber/wireplumber.conf.d/50-force-s16.conf



monitor.alsa.rules = [
  {
    matches = [
      { alsa.id = "ALC892 Analog" }
      { media.class = "Audio/Sink" }
      { node.name = "~alsa_output.*analog.*" }
      { node.name = "~alsa_input.*analog.*" }
      { node.name = "~alsa_input.*" }
      { node.name = "~alsa_output.*" }
    ],
    actions = {
      update-props = {
        "audio.channels" "2"
        "audio.format" = "S16LE"
        "audio.rate" = 44100
        "audio.allowed-rates" = [ 44100 ]
        "resample.disable" = true
        "resample.quality" = 0
        "dither.method" = "Disabled"
        "channelmix.upmix" = false
        "channelmix.upmix-method" = "Disabled"
        "channelmix.mix-lfe" = false
        "channelmix.disable" = true
        "api.alsa.disable-mmap" = true
        "api.alsa.disable-batch" = true
        "api.alsa.disable-tsched" = true
        "api.alsa.use-chmap" = false
        "api.alsa.auto-link" = true
        "api.alsa.multi-rate" = false
        "api.alsa.headroom" = 8192
        "api.alsa.start-delay" = 1024
        "alsa.resolution_bits" = 16
        "api.alsa.period-size" = 3072
        "api.alsa.period-num" = 2
        "audioconvert.filter-graph.disable" = true
      }
    }
  }
]
